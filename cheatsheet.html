Looping in Ruby vs looping in Javascript

Looping with 'while'

Ruby

                while /condition/ do
                  .........
                end

Javascript

    while (/condition/) {
    ...........
  }

Looping certain amount of times

Ruby

                n.times do
                  ..........
                end

Javascript

   for ( var n=x; n /condition/; n++) {
     .........
   }

Iterating through elements of array

Ruby

                array.each do
                  ..........
                end

Javascript

    var a = array;
    for (var i=0; i < a.length; i++) {
      .......................
    }

breaking out of loop

Ruby

                          break

Javascript

                        break

Each method

Ruby

            array.each {|element| code}

Javascript

    var each = function(array,func) {
        for (var i=0; i < array.length; i++) {
        func(array[i]);
      }
     };
      var bugs = [/array/];
      each(bugs,function(x) {console.log(x);});

'map' method

Ruby

                array.map {|element| code}

Javascript

          var results =[];
          for (var i=0; i < array.length; i++) {
            results.push(func(array[i]));
          }
          return results;
            };
            var nums = [5,12,2,32,54];
            var timesTwo=function(x){return x*2};
            var numsTimesTwo = map(nums,timesTwo);

'select' method

Ruby

                   array.select {|element| code}

Javascript

          var select = function(array,func)    {
            var results =[];
            for (var i=0; i < array.length; i++) {
            if (func(array[i])) {
             results.push(func(array[i]));
            }
          }
        return results;
          };

'max' method

Ruby

              array.max

Javascript

            var max = function(array)    {
              var biggest =array[0];
              for (var i=0; i < array.length; i++) {
              if (array[i] > biggest) {
               biggest = array[i]
              }
            }
          return biggest;
            };
