Select method

Ruby

array.select {|element| code}

Javascript

var select = function(array,func) {
  var results =[];
  for (var i=0; i < array.length; i++) {
    if (func(array[i])) {
      results.push(func(array[i]));
    }
  }
  return results;
};

Max Method

Ruby

array.max

Javascript

var max = function(array) {
  var biggest = array[0];
  for (var i=0; i < array.length; i++) {
    if (array[i] > biggest) {
      biggest = array[i]
    }
  }
  return biggest;
};
